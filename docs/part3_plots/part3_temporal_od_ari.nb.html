intraday_trips <- od_all |>
  filter(day_type == "WEEKDAY") |>
  mutate(
    time_band = case_when(
      time_per_hour >= 6  & time_per_hour < 10 ~ "Morning (6-10)",
      time_per_hour >= 10 & time_per_hour < 16 ~ "Afternoon (10-16)",
      time_per_hour >= 16 & time_per_hour < 22 ~ "Evening (16-22)",
      TRUE                                      ~ "Night (22-6)"
    ),
    time_band = factor(time_band, levels = c(
      "Morning (6-10)", "Afternoon (10-16)", "Evening (16-22)", "Night (22-6)"
    ))
  ) |>
  left_join(academic_calendar |> select(ym_key, period), by = "ym_key") |>
  group_by(university, period, time_band) |>
  summarise(total_trips = sum(total_trips), .groups = "drop") |>
  # Convert to share within each university x period
  group_by(university, period) |>
  mutate(share = total_trips / sum(total_trips)) |>
  ungroup()
p_intraday <- ggplot(intraday_trips,
                     aes(x = period, y = share, fill = time_band)) +
  geom_col(width = 0.7) +
  facet_wrap(~ university, nrow = 1) +
  scale_y_continuous(labels = percent_format(accuracy = 1)) +
  scale_fill_manual(
    values = c(
      "Morning (6-10)"    = "#4E79A7",
      "Afternoon (10-16)" = "#F28E2B",
      "Evening (16-22)"   = "#E15759",
      "Night (22-6)"      = "#76B7B2"
    ),
    name = "Time Band"
  ) +
  labs(
    title    = "Intra-day Trip Distribution by University and Academic Period",
    subtitle = "Weekdays only; shares sum to 100% within each university x period",
